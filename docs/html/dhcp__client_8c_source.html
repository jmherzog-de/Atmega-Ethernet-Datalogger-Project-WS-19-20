42 //
43 // The relevant RFCs are
44 // DHCP protocol: http://tools.ietf.org/html/rfc1541
45 // newer version:
46 // DHCP protocol: http://tools.ietf.org/html/rfc2131
47 // message encoding: http://tools.ietf.org/html/rfc1533
48 //
49 // The normal message flow to get an IP address is:
50 // Client -> Server DHCPDISCOVER
51 // Server -> Client DHCPOFFER
52 // Client -> Server DHCPREQUEST
53 // Server -> Client DHCPACK
54 // There might be serveral seconds delay between the DHCPDISCOVER and
55 // the DHCPOFFER as the server does some test (e.g ping or arp) to see
56 // if the IP which is offered is really free.
57 //
58 // At lease renewal the message flow is:
59 // Client -> Server DHCPREQUEST
60 // Server -> Client DHCPACK
61 // The DHCPREQUEST for renewal is different from the one at inital assignment.
62 //
63 //
64 // DHCP_OPTION_OFFSET is a relative to UDP_DATA_P
65 #define DHCP_OPTION_OFFSET 240
66 #define MAGIC_COOKIE_P 236
67 #define DHCP_SRV_SRC_PORT 67
68 #define DHCP_SRV_DST_PORT 68
69 // message type values
70 #define DHCP_DISCOVER_V 0x01
71 #define DHCP_REQUEST_V 0x03

80 // This function writes a basic message template into buf
81 // It processes all fields excluding the options section.
82 // Most fields are initialized with zero.
83 void make_dhcp_message_template(uint8_t *buf,const uint8_t transactionID)
84 {
85  uint8_t i=0;
86  uint8_t allxff[6]={0xff,0xff,0xff,0xff,0xff,0xff}; // all of it can be used as mac, the first 4 can be used as IP
87 
88  send_udp_prepare(buf,DHCP_SRV_DST_PORT,allxff,DHCP_SRV_SRC_PORT,allxff);
89  // source IP is 0.0.0.0, a while loop produces smaller code than memset
90  i=0;
91  while(i<4){
92  buf[IP_SRC_P +i]=0;
93  i++;
94  }
95  // now fill the bootstrap protocol layer starting at UDP_DATA_P
96  buf[UDP_DATA_P]=1;// message type = boot request
97  buf[UDP_DATA_P+1]=1;// hw type
98  buf[UDP_DATA_P+2]=6;// mac len
99  buf[UDP_DATA_P+i+3]=0;// hops
100  // we use only one byte TIDs, we fill the first byte with 1 and
101  // the rest with transactionID. The first byte is used to
102  // distinguish inital requests from renew requests.
103  buf[UDP_DATA_P+4]=1;
104  i=0;
105  while(i<3){
106  buf[UDP_DATA_P+i+5]=transactionID;
107  i++;
108  }
109  // we set seconds always to zero
110  //
111  // set my own MAC the rest is empty:
112  // a while loop produces smaller code than memset
113  i=8; // start at 8
114  while(i<28){
115  buf[UDP_DATA_P +i]=0;
116  i++;
117  }
118  // own mac (send_udp_prepare did fill it at eth level):
119  i=0;
120  while(i<6){
121  buf[UDP_DATA_P+i+28]=buf[ETH_SRC_MAC +i];
122  i++;
123  }
124  // now we need to write 202 bytes of zero
125  // a while loop produces smaller code than memset, don't use memset
126  i=34; // start at 8
127  while(i<236){
128  buf[UDP_DATA_P +i]=0;
129  i++;
130  }
131  // the magic cookie has to be:
132  // 99, 130, 83, 99
133  buf[UDP_DATA_P+MAGIC_COOKIE_P]=0x63;
134  buf[UDP_DATA_P+MAGIC_COOKIE_P+1]=0x82;
135  buf[UDP_DATA_P+MAGIC_COOKIE_P+2]=0x53;
136  buf[UDP_DATA_P+MAGIC_COOKIE_P+3]=0x63;
137 }

140 uint8_t send_dhcp_discover(uint8_t *buf,const uint8_t transactionID)
141 {
142  make_dhcp_message_template(buf,transactionID);
143  // option dhcp message type:
144  buf[UDP_DATA_P+DHCP_OPTION_OFFSET]=0x35; // 53
145  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+1]=1; //len
146  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+2]=DHCP_DISCOVER_V;
147  // option paramter request list:
148  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+3]=0x37; // 55
149  // we want: subnet mask, router
150  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+4]=2; // len
151  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+5]=1; // subnet mask
152  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+6]=3; // router=default GW
153  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+7]=0xff; // end of options
154  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+8]=0;
155  // no padding
156  // the length of the udp message part is now DHCP_OPTION_OFFSET+8
157  send_udp_transmit(buf,DHCP_OPTION_OFFSET+8);
158  return(0);
159 }

161 // scan the options field for the message type field
162 // and return its value.
163 //
164 // Value Message Type
165 // ----- ------------
166 // 1 DHCPDISCOVER
167 // 2 DHCPOFFER (server to client)
168 // 3 DHCPREQUEST
169 // 4 DHCPDECLINE
170 // 5 DHCPACK (server to client)
171 // 6 DHCPNAK (server to client)
172 // 7 DHCPRELEASE
173 // return 0 on message type not found otherwise the numeric
174 // value for the message type as shown in the table above.
175 uint8_t dhcp_get_message_type(uint8_t *buf,uint16_t plen)
176 {
177  uint16_t option_idx;
178  uint8_t option_len;
179  // the smallest option is 3 bytes
180  if (plen<(UDP_DATA_P+DHCP_OPTION_OFFSET+3)) return(0);
181  // options are coded in the form: option_type,option_len,option_val
182  option_idx=UDP_DATA_P+DHCP_OPTION_OFFSET;
183  while(option_idx+2 <plen ){
184  option_len=buf[option_idx+1];
185  if ((option_len<1) || ((option_idx + option_len + 1)> plen)) break;
186  if (buf[option_idx]==53){
187  // found message type, return it:
188  return(buf[option_idx+2]);
189  }
190  option_idx+=2+option_len;
191  }
192  return(0);
193 }

229 uint8_t dhcp_option_parser(uint8_t *buf,uint16_t plen)
230 {
231  uint16_t option_idx;
232  uint8_t option_len;
233  uint8_t i;
234  uint32_t ltime_int32=0;
235  // the smallest option is 3 bytes
236  if (plen<(UDP_DATA_P+DHCP_OPTION_OFFSET+3)) return(0);
237  // options are coded in the form: option_type,option_len,option_val
238  option_idx=UDP_DATA_P+DHCP_OPTION_OFFSET;
239  while(option_idx+2 <plen ){
240  option_len=buf[option_idx+1];
241  if ((option_len<1) || ((option_idx + option_len + 1)> plen)) break;
242  switch (buf[option_idx]){
243  case 0: option_idx=plen; // stop loop, we are reading some padding bytes here (should not happen)
244  break;
245 
246  case 1: if (option_len==4){
247  i=0;
248  while(i<4){
249  dhcp_opt_mask[i]=buf[option_idx+2+i];
250  i++;
251  }
252  }
253  break;
254  case 3: if (option_len==4){
255  i=0;
256  while(i<4){
257  dhcp_opt_defaultgw[i]=buf[option_idx+2+i];
258  i++;
259  }
260  }
261  break;
262  // Lease time: throughout the protocol, times are to
263  // be represented in units of seconds. The time value
264  // of 0xffffffff is reserved to represent "infinity".
265  // The max lease time size is therefore 32 bit.
266  // The code for this option is 51, and its length is 4
267  // as per RFC 1533.
268  case 51:
269  if (option_len!=4){
270  dhcp_opt_leasetime_minutes=1440; // just to have a reasonable value: 1 day
271  break;
272  }
273  i=0;
274  while(i<4){
275  ltime_int32=(ltime_int32<<8) | buf[option_idx+i+2];
276  i++;
277  }
278  if (ltime_int32==0xffffffff){
279  // lease time is infinity
280  dhcp_opt_leasetime_minutes=0xffff;
281  break; // end of switch
282  }
283  ltime_int32=ltime_int32>>6; // an inexpesive way to divide by 64 (which is roughly equal to divide by 60)
284  if (ltime_int32> 0xfffd){ //0xffff is not handeled here because of the above break
285  dhcp_opt_leasetime_minutes=0xfffd;
286  }else{
287  dhcp_opt_leasetime_minutes=ltime_int32 & 0xffff;
288  }
289  if (dhcp_opt_leasetime_minutes<5) dhcp_opt_leasetime_minutes=5;
290  break;
291  case 53: dhcp_opt_message_type=buf[option_idx+2];
292  break;
293  // rfc 2131: A DHCP server always returns its
294  // own address in the 'server identifier' option
295  case 54: if (option_len==4){
296  i=0;
297  while(i<4){
298  dhcp_opt_server_id[i]=buf[option_idx+2+i];
299  i++;
300  }
301  }
302  break;
303  }
304  option_idx+=2+option_len;
305  }
306  return(1);
307 }

310 uint8_t send_dhcp_request(uint8_t *buf,const uint8_t transactionID)
311 {
312  uint8_t i=0;
313  make_dhcp_message_template(buf,transactionID);
314  // option dhcp message type:
315  buf[UDP_DATA_P+DHCP_OPTION_OFFSET]=0x35; // 53
316  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+1]=1; //len
317  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+2]=DHCP_REQUEST_V;
318  i=3;
319  if (dhcp_opt_server_id[0]!=0){
320  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i]=0x36; // 54=server identifier
321  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i+1]=4; // len
322  memcpy(buf+UDP_DATA_P+DHCP_OPTION_OFFSET+i+2,dhcp_opt_server_id, 4);
323  i=i+6;
324  }
325  if (dhcp_yiaddr[0]!=0){
326  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i]=0x32; // 50=requested IP address
327  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i+1]=4; // len
328  memcpy(buf+UDP_DATA_P+DHCP_OPTION_OFFSET+i+2,dhcp_yiaddr, 4);
329  i=i+6;
330  }
331  // option paramter request list:
332  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i]=0x37; // 55
333  // we want: subnet mask, router
334  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i+1]=2; // len
335  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i+2]=1; // subnet mask
336  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i+3]=3; // router=default GW
337  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i+4]=0xff; // end of options
338  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+i+5]=0;
339  // the length of the udp message part is now DHCP_OPTION_OFFSET+i+5
340  send_udp_transmit(buf,DHCP_OPTION_OFFSET+i+5);
341  return(0);
342 }

344 // The renew procedure is described in rfc2131.
345 // We send DHCPREQUEST and 'server identifier' MUST NOT be filled
346 // in, 'requested IP address' option MUST NOT be filled in, 'ciaddr'
347 // MUST be filled.
348 // The rfc suggest that I can send the DHCPREQUEST in this case as
349 // a unicast message and not as a broadcast message but test with
350 // various DHCP servers show that not all of them listen to
351 // unicast. We send therefor a broadcast message but we expect
352 // a unicast answer directly to our mac and IP.
353 uint8_t send_dhcp_renew_request(uint8_t *buf,const uint8_t transactionID,uint8_t *yiaddr)
354 {
355  make_dhcp_message_template(buf,transactionID);
356  buf[UDP_DATA_P+4]=2; // set first byte in transaction ID to 2 to indicate renew_request. This trick makes the processing of the DHCPACK message easier.
357  // source IP must be my IP since we renew
358  memcpy(buf+IP_SRC_P, yiaddr, 4); // ip level source IP
359  //
360  memcpy(buf+UDP_DATA_P+12,yiaddr, 4); // ciaddr
361  // option dhcp message type:
362  buf[UDP_DATA_P+DHCP_OPTION_OFFSET]=0x35; // 53
363  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+1]=1; //len
364  buf[UDP_DATA_P+DHCP_OPTION_OFFSET+2]=DHCP_REQUEST_V;
365  // no option paramter request list is needed at renew
366  send_udp_transmit(buf,DHCP_OPTION_OFFSET+3);
367  // we will get a unicast answer, reception of broadcast packets is turned off
368  return(0);
369 }

372 // Initial_tid can be a random number for every board. E.g the last digit
373 // of the mac address. It is not so important that the number is random.
374 // It is more important that it is unique and no other board on the same
375 // Lan has the same number. This is because there might be a power outage
376 // and all boards reboot afterwards at the same time. At that moment they
377 // must all have different TIDs otherwise there will be an IP address mess-up.
378 //
379 // The function returns 1 once we have a valid IP.
380 // At this point you must not call the function again.
381 uint8_t packetloop_dhcp_initial_ip_assignment(uint8_t *buf,uint16_t plen,uint8_t initial_tid){
382  static uint16_t init=0x7fff; // about 3 sec delay
383  uint8_t cmd;
384  // do nothing if the link is down
385  if (!enc28j60linkup()) return(0);
386  // first time that this function is called:
387  if (plen==0){
388  if (init==2){
389  init=1;
390  dhcp_6sec_cnt=0;
391  dhcp_tid=initial_tid;
392  // Reception of broadcast packets is turned off by default, but
393  // the DHCP offer message that the DHCP server sends will be
394  // a broadcast packet. Enable here and disable later.
395  enc28j60EnableBroadcast();
396  send_dhcp_discover(buf,dhcp_tid);
397  return(0);
398  }
399  if (dhcp_yiaddr[0]==0 && dhcp_6sec_cnt > 5){
400  // still no IP after 30 sec
401  dhcp_tid++;
402  dhcp_6sec_cnt=0;
403  // Reception of broadcast packets is turned off by default, but
404  // the DHCP offer message that the DHCP server sends will be
405  // a broadcast packet. Enable here and disable later.
406  enc28j60EnableBroadcast();
407  send_dhcp_discover(buf,dhcp_tid);
408  return(0);
409  }
410  // We have a little delay (about 3sec) at startup. Sometimes
411  // the power fluctuates or the programmer causes
412  // the board to reset and then immediately reset again.
413  // We wait with the sending of a send_dhcp_discover just in case
414  // we did already so at the last reset which was possibly less
415  // than a second ago.
416  if (init>2){
417  init--;
418  }
419  return(0);
420  }
421  // plen > 0
422  if (is_dhcp_msg_for_me(buf,plen,dhcp_tid)){
423  // It's really a borderline case that we the the dhcp_is_renew_tid
424  // function call for. It could only happen if the board is power cyceled
425  // during operation.
426  if (dhcp_is_renew_tid(buf,plen)==1) return(0); // should have been initial tid, just return
427  cmd=dhcp_get_message_type(buf,plen);
428  if (cmd==2){ // DHCPOFFER =2
429  init=1; // no more init needed
430  dhcp_get_yiaddr(buf,plen);
431  dhcp_option_parser(buf,plen);
432  // answer offer with a request:
433  send_dhcp_request(buf,dhcp_tid);
434  }
435  if (cmd==5){ // DHCPACK =5
436  // success, DHCPACK, we have the IP
437  init=1; // no more init needed
438  enc28j60DisableBroadcast();
439  return(1);
440  }
441  }
442  return(0);
443 }

457 // returns plen of original packet if buf is not touched.
458 // returns 0 if plen was originally zero. returns 0 if DHCP messages
459 // was processed.
460 // We don't need to expect changing IP addresses. We can stick
461 // to the IP that we got once. The server has really no power to

463 uint16_t packetloop_dhcp_renewhandler(uint8_t *buf,uint16_t plen){
464  if (dhcp_6sec_cnt> 8){ // we let it run a bit faster than once every minute because it is better this expires too early than too late
465  dhcp_6sec_cnt=0;
466  // count down unless the lease was infinite
467  if (dhcp_opt_leasetime_minutes < 0xffff && dhcp_opt_leasetime_minutes>1){
468  dhcp_opt_leasetime_minutes--;
469  }
470  }
471  if (plen ==0 && dhcp_opt_leasetime_minutes < 3){
472  if (!enc28j60linkup()) return(plen); // do nothing if link is down
473  dhcp_tid++;
474  send_dhcp_renew_request(buf,dhcp_tid,dhcp_yiaddr);
475  dhcp_opt_leasetime_minutes=5; // repeat in two minutes if no answer
476  return(0);
477  }
478  if (plen && is_dhcp_msg_for_me(buf,plen,dhcp_tid)){
479  // we check the dhcp_renew_tid because if
480  if (dhcp_get_message_type(buf,plen)==5){ // DHCPACK =5
481  // success, DHCPACK, we have the IP
482  if (dhcp_is_renew_tid(buf,plen)){
483  dhcp_option_parser(buf,plen); // get new lease time, it will as well GW and netmask but those should not change
484  }
485  }
486  return(0);
487  }
488  return(plen);
489 }
