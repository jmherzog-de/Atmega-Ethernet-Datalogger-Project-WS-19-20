72 // send a DNS udp request packet
73 // See http://www.ietf.org/rfc/rfc1034.txt
74 // and http://www.ietf.org/rfc/rfc1035.txt
75 // gwmac is the internal mac addess of your router
76 // because we use 8.8.8.8 as a DNS server
77 void dnslkup_request(uint8_t *buf,const char *hostname,const uint8_t *gwmac)
78 {
79  uint8_t i,lenpos,lencnt;
80  haveDNSanswer=0;
81  dns_ansError=0;
82  dnstid_l++; // increment for next request, finally wrap
83  send_udp_prepare(buf,(DNSCLIENT_SRC_PORT_H<<8)|(dnstid_l&0xff),dnsip,53,gwmac);
84  // fill tid:
85  //buf[UDP_DATA_P] see below
86  buf[UDP_DATA_P+1]=dnstid_l;
87  buf[UDP_DATA_P+2]=1; // flags, standard recursive query
88  i=3;
89  // most fields are zero, here we zero everything and fill later
90  while(i<12){
91  buf[UDP_DATA_P+i]=0;
92  i++;
93  }
94  buf[UDP_DATA_P+5]=1; // 1 question
95  // the structure of the domain name
96  // we ask for is always length, string, length, string, ...
97  // for earch portion of the name.
98  // www.twitter.com would become: 3www7twitter3com
99  //
100  // the first len starts at i=12
101  lenpos=12;
102  i=13;
103  lencnt=1; // need to start with one as there is no dot before the domain name and the below algorithm assumes lencnt=0 at dot
104  while(*hostname){
105  if (*hostname=='\0') break;
106  if (*hostname=='.'){
107  buf[UDP_DATA_P+lenpos]=lencnt-1; // fill the length field
108  lencnt=0;
109  lenpos=i;
110  }
111  buf[UDP_DATA_P+i]=*hostname;
112  lencnt++;
113  i++;
114  hostname++;
115  }
116  buf[UDP_DATA_P+lenpos]=lencnt-1;
117  buf[UDP_DATA_P+i]=0; // terminate with zero, means root domain.
118  i++;
119  buf[UDP_DATA_P+i]=0;
120  i++;
121  buf[UDP_DATA_P+i]=1; // type A
122  i++;
123  buf[UDP_DATA_P+i]=0;
124  i++;
125  buf[UDP_DATA_P+i]=1; // class IN
126  i++;
127  // we encode the length into the upper byte of the TID
128  // this way we can easily jump over the query section
129  // of the answer:
130  buf[UDP_DATA_P]=i-12;
131  send_udp_transmit(buf,i);
132 }

134 // process the answer from the dns server:
135 // return 1 on sucessful processing of answer.
136 // We set also the variable haveDNSanswer
137 uint8_t udp_client_check_for_dns_answer(uint8_t *buf,uint16_t plen){
138  uint8_t j,i;
139  if (plen<70){
140  return(0);
141  }
142  if (buf[UDP_SRC_PORT_L_P]!=53){
143  // not from a DNS
144  return(0);
145  }
146  if (buf[UDP_DST_PORT_H_P]!=DNSCLIENT_SRC_PORT_H){
147  return(0);
148  }
149  // we use the TID also as port:
150  if (buf[UDP_DST_PORT_L_P]!=dnstid_l){
151  return(0);
152  }
153  /* we can skip this check, it is quite unlikely that we
154  * get a packet with the right tid from a DNS. Save some
155  * processing:
156  // is the packet for my IP:
157  if(eth_type_is_ip_and_my_ip(buf,plen)==0){
158  return(0);
159  }
160  */
161  if (buf[UDP_DATA_P+1]!=dnstid_l){
162  return(0);
163  }
164  // check flags lower byte:
165  if ((buf[UDP_DATA_P+3]&0x8F)!=0x80){
166  // there is an error or server does not support recursive
167  // queries. We can only work with servers that support recursive
168  // queries.
169  dns_ansError=1;
170  return(0);
171  }
172  // there might be multiple answers, we use only the first one
173  //
174  // UDP_DATA_P+12+querylen is first byte of first answer.
175  // The answer contains again the domain name and we need to
176  // jump over it to find the IP. This part can be abbreviated by
177  // the use of 2 byte pointers. See RFC 1035.
178  i=12+buf[UDP_DATA_P]; // we encoded the query len into tid
179 ChecNextResp:
180  if (buf[UDP_DATA_P+i] & 0xc0){
181  // pointer
182  i+=2;
183  }else{
184  // we just search for the first, zero=root domain
185  // all other octets must be non zero
186  while(i<plen-UDP_DATA_P-7){
187  i++;
188  if (buf[UDP_DATA_P+i]==0){
189  i++;
190  break;
191  }
192  }
193  }
194  // There might be multipe records in the answer.
195  // We are searching for an A record (contains IP).
196  // Contributed by Andras Tucsni
197  if (buf[UDP_DATA_P+i+1] != 1){ // check type == 1 for "A"
198  i += 2 + 2 + 4; // skip type & class & TTL
199  i += buf[UDP_DATA_P+i+1] + 2; // skip datalength bytes
200  if (i < plen-UDP_DATA_P-7){
201  goto ChecNextResp;
202  }
203  dns_ansError=3; // no A record found but packet ends
204  return(0);
205  }
206  // We have an A record
207  // i is now pointing to the high octet of the type field
208  if (buf[UDP_DATA_P+i+9] !=4 ){
209  dns_ansError=2; // not IPv4
210  return(0);
211  }
212  i+=10;
213  j=0;
214  while(j<4){
215  dns_answerip[j]=buf[UDP_DATA_P+i+j];
216  j++;
217  }
218  haveDNSanswer=1;
219  return(1);
220 }
