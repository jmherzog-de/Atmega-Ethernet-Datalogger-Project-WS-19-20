48 uint8_t enc28j60ReadOp(uint8_t op, uint8_t address)
49 {
50  CSACTIVE;
51  // issue read command
52  SPDR = op | (address & ADDR_MASK);
53  waitspi();
54  // read data
55  SPDR = 0x00;
56  waitspi();
57  // do dummy read if needed (for mac and mii, see datasheet page 29)
58  if(address & 0x80)
59  {
60  SPDR = 0x00;
61  waitspi();
62  }
63  // release CS
64  CSPASSIVE;
65  return(SPDR);
66 }

80 void enc28j60ReadBuffer(uint16_t len, uint8_t* data)
81 {
82  CSACTIVE;
83  // issue read command
84  SPDR = ENC28J60_READ_BUF_MEM;
85  waitspi();
86  while(len)
87  {
88  len--;
89  // read data
90  SPDR = 0x00;
91  waitspi();
92  *data = SPDR;
93  data++;
94  }

96  CSPASSIVE;

116 void enc28j60SetBank(uint8_t address)
117 {
118  // set the bank (if needed)
119  if((address & BANK_MASK) != Enc28j60Bank)
120  {
121  // set the bank
122  enc28j60WriteOp(ENC28J60_BIT_FIELD_CLR, ECON1, (ECON1_BSEL1|ECON1_BSEL0));
123  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, ECON1, (address & BANK_MASK)>>5);
124  Enc28j60Bank = (address & BANK_MASK);

181 void enc28j60Init(uint8_t* macaddr)
182 {
183  // initialize I/O
184  // ss as output:
185  /*ENC28J60_CONTROL_DDR |= 1<<ENC28J60_CONTROL_CS;
186  CSPASSIVE; // ss=0
187  //
188  ENC28J60_CONTROL_DDR |= 1<<ENC28J60_CONTROL_SI | 1<<ENC28J60_CONTROL_SCK; // mosi, sck output
189  ENC28J60_CONTROL_DDR|= 1<<ENC28J60_CONTROL_SO; // MISO is input
190  //
191  ENC28J60_CONTROL_PORT|= 1<<ENC28J60_CONTROL_SI; // MOSI low
192  ENC28J60_CONTROL_PORT|= 1<<ENC28J60_CONTROL_SCK; // SCK low
193  //
194  // initialize SPI interface
195  // master mode and Fosc/2 clock:
196  SPCR = (1<<SPE) | (1<<MSTR) | (1<<SPR0);
197  //SPSR |= (1<<SPI2X);*/
198  // perform system reset
199  enc28j60WriteOp(ENC28J60_SOFT_RESET, 0, ENC28J60_SOFT_RESET);
200  _delay_loop_2(0); // 20ms
201  // check CLKRDY bit to see if reset is complete
202  // The CLKRDY does not work. See Rev. B4 Silicon Errata point. Just wait.
203  //while(!(enc28j60Read(ESTAT) & ESTAT_CLKRDY));
204  // do bank 0 stuff
205  // initialize receive buffer
206  // 16-bit transfers, must write low byte first
207  // set receive buffer start address
208  gNextPacketPtr = RXSTART_INIT;
209  // Rx start
210  enc28j60Write(ERXSTL, RXSTART_INIT&0xFF);
211  enc28j60Write(ERXSTH, RXSTART_INIT>>8);
212  // set receive pointer address
213  enc28j60Write(ERXRDPTL, RXSTART_INIT&0xFF);
214  enc28j60Write(ERXRDPTH, RXSTART_INIT>>8);
215  // RX end
216  enc28j60Write(ERXNDL, RXSTOP_INIT&0xFF);
217  enc28j60Write(ERXNDH, RXSTOP_INIT>>8);
218  // TX start
219  enc28j60Write(ETXSTL, TXSTART_INIT&0xFF);
220  enc28j60Write(ETXSTH, TXSTART_INIT>>8);
221  // TX end
222  enc28j60Write(ETXNDL, TXSTOP_INIT&0xFF);
223  enc28j60Write(ETXNDH, TXSTOP_INIT>>8);
224  // do bank 1 stuff, packet filter:
225  // For broadcast packets we allow only ARP packtets
226  // All other packets should be unicast only for our mac (MAADR)
227  //
228  // The pattern to match on is therefore
229  // Type ETH.DST
230  // ARP BROADCAST
231  // 06 08 -- ff ff ff ff ff ff -> ip checksum for theses bytes=f7f9
232  // in binary these poitions are:11 0000 0011 1111
233  // This is hex 303F->EPMM0=0x3f,EPMM1=0x30
234  enc28j60Write(ERXFCON, ERXFCON_UCEN|ERXFCON_CRCEN|ERXFCON_PMEN);
235  enc28j60Write(EPMM0, 0x3f);
236  enc28j60Write(EPMM1, 0x30);
237  enc28j60Write(EPMCSL, 0xf9);
238  enc28j60Write(EPMCSH, 0xf7);
239  //
240  //
241  // do bank 2 stuff
242  // enable MAC receive
243  enc28j60Write(MACON1, MACON1_MARXEN|MACON1_TXPAUS|MACON1_RXPAUS);
244  // bring MAC out of reset
245  enc28j60Write(MACON2, 0x00);
246  // enable automatic padding to 60bytes and CRC operations
247  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, MACON3, MACON3_PADCFG0|MACON3_TXCRCEN|MACON3_FRMLNEN);
248  // set inter-frame gap (non-back-to-back)
249  enc28j60Write(MAIPGL, 0x12);
250  enc28j60Write(MAIPGH, 0x0C);
251  // set inter-frame gap (back-to-back)
252  enc28j60Write(MABBIPG, 0x12);
253  // Set the maximum packet size which the controller will accept
254  // Do not send packets longer than MAX_FRAMELEN:
255  enc28j60Write(MAMXFLL, MAX_FRAMELEN&0xFF);
256  enc28j60Write(MAMXFLH, MAX_FRAMELEN>>8);
257  // do bank 3 stuff
258  // write MAC address
259  // NOTE: MAC address in ENC28J60 is byte-backward
260  enc28j60Write(MAADR5, macaddr[0]);
261  enc28j60Write(MAADR4, macaddr[1]);
262  enc28j60Write(MAADR3, macaddr[2]);
263  enc28j60Write(MAADR2, macaddr[3]);
264  enc28j60Write(MAADR1, macaddr[4]);
265  enc28j60Write(MAADR0, macaddr[5]);
266  // no loopback of transmitted frames
267  enc28j60PhyWrite(PHCON2, PHCON2_HDLDIS);
268  // switch to bank 0
269  enc28j60SetBank(ECON1);
270  // enable interrutps
271  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, EIE, EIE_INTIE|EIE_PKTIE);
272  // enable packet reception
273  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, ECON1, ECON1_RXEN);
274 }

311 void enc28j60PacketSend(uint16_t len, uint8_t* packet)
312 {
313  // Check no transmit in progress
314  while (enc28j60ReadOp(ENC28J60_READ_CTRL_REG, ECON1) & ECON1_TXRTS)
315  {
316  // Reset the transmit logic problem. See Rev. B4 Silicon Errata point 12.
317  if( (enc28j60Read(EIR) & EIR_TXERIF) ) {
318  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, ECON1, ECON1_TXRST);
319  enc28j60WriteOp(ENC28J60_BIT_FIELD_CLR, ECON1, ECON1_TXRST);
320  }
321  }
322  // Set the write pointer to start of transmit buffer area
323  enc28j60Write(EWRPTL, TXSTART_INIT&0xFF);
324  enc28j60Write(EWRPTH, TXSTART_INIT>>8);
325  // Set the TXND pointer to correspond to the packet size given
326  enc28j60Write(ETXNDL, (TXSTART_INIT+len)&0xFF);
327  enc28j60Write(ETXNDH, (TXSTART_INIT+len)>>8);
328  // write per-packet control byte (0x00 means use macon3 settings)
329  enc28j60WriteOp(ENC28J60_WRITE_BUF_MEM, 0, 0x00);
330  // copy the packet into the transmit buffer
331  enc28j60WriteBuffer(len, packet);
332  // send the contents of the transmit buffer onto the network
333  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, ECON1, ECON1_TXRTS);
334 }

345 // Gets a packet from the network receive buffer, if one is available.
346 // The packet will by headed by an ethernet header.
347 // maxlen The maximum acceptable length of a retrieved packet.
348 // packet Pointer where packet data should be stored.
349 // Returns: Packet length in bytes if a packet was retrieved, zero otherwise.
350 uint16_t enc28j60PacketReceive(uint16_t maxlen, uint8_t* packet)
351 {
352  uint16_t rxstat;
353  uint16_t len;
354  // check if a packet has been received and buffered
355  //if( !(enc28j60Read(EIR) & EIR_PKTIF) )
356  // The above does not work. See Rev. B4 Silicon Errata point 6.
357  if( enc28j60Read(EPKTCNT) ==0 ){
358  return(0);
359  }
360 
361  // Set the read pointer to the start of the received packet
362  enc28j60Write(ERDPTL, (gNextPacketPtr &0xFF));
363  enc28j60Write(ERDPTH, (gNextPacketPtr)>>8);
364  // read the next packet pointer
365  gNextPacketPtr = enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0);
366  gNextPacketPtr |= enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0)<<8;
367  // read the packet length (see datasheet page 43)
368  len = enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0);
369  len |= enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0)<<8;
370  len-=4; //remove the CRC count
371  // read the receive status (see datasheet page 43)
372  rxstat = enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0);
373  rxstat |= ((uint16_t)enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0))<<8;
374  // limit retrieve length
375  if (len>maxlen-1){
376  len=maxlen-1;
377  }
378  // check CRC and symbol errors (see datasheet page 44, table 7-3):
379  // The ERXFCON.CRCEN is set by default. Normally we should not
380  // need to check this.
381  if ((rxstat & 0x80)==0){
382  // invalid
383  len=0;
384  }else{
385  // copy the packet from the receive buffer
386  enc28j60ReadBuffer(len, packet);
387  }
388  // Move the RX read pointer to the start of the next received packet
389  // This frees the memory we just read out
390  //enc28j60Write(ERXRDPTL, (gNextPacketPtr &0xFF));
391  //enc28j60Write(ERXRDPTH, (gNextPacketPtr)>>8);
392  //
393  // Move the RX read pointer to the start of the next received packet
394  // This frees the memory we just read out.
395  // However, compensate for the errata point 13, rev B4: never write an even address!
396  // gNextPacketPtr is always an even address if RXSTOP_INIT is odd.
397  if (gNextPacketPtr -1 > RXSTOP_INIT){ // RXSTART_INIT is zero, no test for gNextPacketPtr less than RXSTART_INIT.
398  enc28j60Write(ERXRDPTL, (RXSTOP_INIT)&0xFF);
399  enc28j60Write(ERXRDPTH, (RXSTOP_INIT)>>8);
400  } else {
401  enc28j60Write(ERXRDPTL, (gNextPacketPtr-1)&0xFF);
402  enc28j60Write(ERXRDPTH, (gNextPacketPtr-1)>>8);
403  }
404  // decrement the packet counter indicate we are done with this packet
405  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, ECON2, ECON2_PKTDEC);
406  return(len);
407 }
