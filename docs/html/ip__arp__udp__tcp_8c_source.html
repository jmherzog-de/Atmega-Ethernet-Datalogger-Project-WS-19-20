111 // The Ip checksum is calculated over the ip header only starting
112 // with the header length field and a total length of 20 bytes
113 // unitl ip.dst
114 // You must set the IP checksum field to zero before you start
115 // the calculation.
116 // len for ip is 20.
117 //
118 // For UDP/TCP we do not make up the required pseudo header. Instead we
119 // use the ip.src and ip.dst fields of the real packet:
120 // The udp checksum calculation starts with the ip.src field
121 // Ip.src=4bytes,Ip.dst=4 bytes,Udp header=8bytes + data length=16+len
122 // In other words the len here is 8 + length over which you actually
123 // want to calculate the checksum.
124 // You must set the checksum field to zero before you start
125 // the calculation.
126 // The same algorithm is also used for udp and tcp checksums.
127 // len for udp is: 8 + 8 + data length
128 // len for tcp is: 4+4 + 20 + option len + data length
129 //
130 // For more information on how this algorithm works see:
131 // http://www.netfor2.com/checksum.html
132 // http://www.msc.uky.edu/ken/cs471/notes/chap3.htm
133 // The RFC has also a C code example: http://www.faqs.org/rfcs/rfc1071.html
134 uint16_t checksum(uint8_t *buf, uint16_t len,uint8_t type){
135  // type 0=ip , icmp
136  // 1=udp
137  // 2=tcp
138  uint32_t sum = 0;
139 
140  //if(type==0){
141  // // do not add anything, standard IP checksum as described above
142  // // Usable for ICMP and IP header
143  //}
144  if(type==1){
145  sum+=IP_PROTO_UDP_V; // protocol udp
146  // the length here is the length of udp (data+header len)
147  // =length given to this function - (IP.scr+IP.dst length)
148  sum+=len-8; // = real udp len
149  }
150  if(type==2){
151  sum+=IP_PROTO_TCP_V;
152  // the length here is the length of tcp (data+header len)
153  // =length given to this function - (IP.scr+IP.dst length)
154  sum+=len-8; // = real tcp len
155  }
156  // build the sum of 16bit words
157  while(len >1){
158  sum += 0xFFFF & (((uint32_t)*buf<<8)|*(buf+1));
159  buf+=2;
160  len-=2;
161  }
162  // if there is a byte left then add it (padded with zero)
163  if (len){
164  sum += ((uint32_t)(0xFF & *buf))<<8;
165  }
166  // now calculate the sum over the bytes in the sum
167  // until the result is only 16bit long
168  while (sum>>16){
169  sum = (sum & 0xFFFF)+(sum >> 16);
170  }
171  // build 1's complement:
172  return( (uint16_t) sum ^ 0xFFFF);
173 }

324 // make a return tcp header from a received tcp packet
325 // rel_ack_num is how much we must step the seq number received from the
326 // other side. We do not send more than 765 bytes of text (=data) in the tcp packet.
327 // No mss is included here.
328 //
329 // After calling this function you can fill in the first data byte at TCP_OPTIONS_P+4
330 // If cp_seq=0 then an initial sequence number is used (should be use in synack)
331 // otherwise it is copied from the packet we received
332 void make_tcphead(uint8_t *buf,uint16_t rel_ack_num,uint8_t cp_seq)
333 {
334  uint8_t i;
335  // copy ports:
336  i=buf[TCP_DST_PORT_H_P];
337  buf[TCP_DST_PORT_H_P]=buf[TCP_SRC_PORT_H_P];
338  buf[TCP_SRC_PORT_H_P]=i;
339  //
340  i=buf[TCP_DST_PORT_L_P];
341  buf[TCP_DST_PORT_L_P]=buf[TCP_SRC_PORT_L_P];
342  buf[TCP_SRC_PORT_L_P]=i;
343  step_seq(buf,rel_ack_num,cp_seq);
344  // zero the checksum
345  buf[TCP_CHECKSUM_H_P]=0;
346  buf[TCP_CHECKSUM_L_P]=0;
347  // no options:
348  // 20 bytes:
349  // The tcp header length is only a 4 bit field (the upper 4 bits).
350  // It is calculated in units of 4 bytes.
351  // E.g 20 bytes: 20/4=6 => 0x50=header len field
352  buf[TCP_HEADER_LEN_P]=0x50;
353 }
