84 #ifdef TCP_client
85 // To use the tcp client you need to:
86 //
87 // Declare a callback function to get the result (tcp data from the server):
88 //
89 // uint8_t your_client_tcp_result_callback(uint8_t fd, uint8_t statuscode,uint16_t data_start_pos_in_buf, uint16_t len_of_data){
90 // ...your code;
91 // return(close_tcp_session);
92 // }
93 //
94 // statuscode=0 means the buffer has valid data, otherwise len and pos_in_buf
95 // are invalid. That is: do to use data_start_pos_in_buf and len_of_data
96 // if statuscode!=0.
97 //
98 // This callback gives you access to the TCP data of the first
99 // packet returned from the server. You should aim to minimize the server
100 // output such that this will be the only packet.
101 //
102 // close_tcp_session=1 means close the session now. close_tcp_session=0
103 // read all data and leave it to the other side to close it.
104 // If you connect to a web server then you want close_tcp_session=0.
105 // If you connect to a modbus/tcp equipment then you want close_tcp_session=1
106 //
107 // Declare a callback function to be called in order to fill in the
108 //
109 // request (tcp data sent to the server):
110 // uint16_t your_client_tcp_datafill_callback(uint8_t fd){...your code;return(len_of_data_filled_in);}
111 //
112 // Now call:
113 // fd=client_tcp_req(&your_client_tcp_result_callback,&your_client_tcp_datafill_callback,portnumber);
114 //
115 // fd is a file descriptor like number that you get back in the fill and result
116 // function so you know to which call of client_tcp_req this callback belongs.
117 //
118 // You can not start different clients (e.g modbus and web) at the
119 // same time but you can start them one after each other. That is
120 // when the request has timed out or when the result_callback was
121 // executed then you can start a new one. The fd makes it still possible to
122 // distinguish in the callback code the different types you started.
123 //
124 // Note that you might never get called back if the other side does
125 // not answer. A timer would be needed to recongnize such a condition.
126 //
127 // We use callback functions because that is the best implementation
128 // given the fact that we have very little RAM memory.
129 //
130 extern uint8_t client_tcp_req(uint8_t (*result_callback)(uint8_t fd,uint8_t statuscode,uint16_t data_start_pos_in_buf, uint16_t len_of_data),uint16_t (*datafill_callback)(uint8_t fd),uint16_t port,uint8_t *dstip,uint8_t *dstmac);
